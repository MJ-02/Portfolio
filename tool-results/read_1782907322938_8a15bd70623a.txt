     1→'use client'
     2→
     3→import React, { useCallback, useRef, useEffect, useState } from 'react'
     4→import { useXPStore, WindowState } from '@/store/xp-store'
     5→
     6→interface XPWindowProps {
     7→  window: WindowState
     8→}
     9→
    10→export default function XPWindow({ window: win }: XPWindowProps) {
    11→  const {
    12→    closeWindow,
    13→    minimizeWindow,
    14→    maximizeWindow,
    15→    restoreWindow,
    16→    focusWindow,
    17→    moveWindow,
    18→    resizeWindow,
    19→    activeWindowId,
    20→  } = useXPStore()
    21→
    22→  const isActive = activeWindowId === win.id
    23→  const dragRef = useRef<{ startX: number; startY: number; winX: number; winY: number } | null>(null)
    24→  const resizeRef = useRef<{ startX: number; startY: number; winW: number; winH: number } | null>(null)
    25→  const windowRef = useRef<HTMLDivElement>(null)
    26→
    27→  const handleMouseDown = useCallback((e: React.MouseEvent) => {
    28→    if (win.maximized) return
    29→    e.preventDefault()
    30→    focusWindow(win.id)
    31→    dragRef.current = {
    32→      startX: e.clientX,
    33→      startY: e.clientY,
    34→      winX: win.x,
    35→      winY: win.y,
    36→    }
    37→  }, [win.id, win.x, win.y, win.maximized, focusWindow])
    38→
    39→  const handleResizeStart = useCallback((e: React.MouseEvent) => {
    40→    if (win.maximized) return
    41→    e.preventDefault()
    42→    e.stopPropagation()
    43→    focusWindow(win.id)
    44→    resizeRef.current = {
    45→      startX: e.clientX,
    46→      startY: e.clientY,
    47→      winW: win.width,
    48→      winH: win.height,
    49→    }
    50→  }, [win.id, win.width, win.height, win.maximized, focusWindow])
    51→
    52→  useEffect(() => {
    53→    const handleMouseMove = (e: MouseEvent) => {
    54→      if (dragRef.current) {
    55→        const dx = e.clientX - dragRef.current.startX
    56→        const dy = e.clientY - dragRef.current.startY
    57→        const newX = Math.max(0, dragRef.current.winX + dx)
    58→        const newY = Math.max(0, dragRef.current.winY + dy)
    59→        moveWindow(win.id, newX, newY)
    60→      }
    61→      if (resizeRef.current) {
    62→        const dx = e.clientX - resizeRef.current.startX
    63→        const dy = e.clientY - resizeRef.current.startY
    64→        const newW = Math.max(win.minWidth, resizeRef.current.winW + dx)
    65→        const newH = Math.max(win.minHeight, resizeRef.current.winH + dy)
    66→        resizeWindow(win.id, newW, newH)
    67→      }
    68→    }
    69→
    70→    const handleMouseUp = () => {
    71→      dragRef.current = null
    72→      resizeRef.current = null
    73→    }
    74→
    75→    window.addEventListener('mousemove', handleMouseMove)
    76→    window.addEventListener('mouseup', handleMouseUp)
    77→    return () => {
    78→      window.removeEventListener('mousemove', handleMouseMove)
    79→      window.removeEventListener('mouseup', handleMouseUp)
    80→    }
    81→  }, [win.id, win.minWidth, win.minHeight, moveWindow, resizeWindow])
    82→
    83→  if (win.minimized) return null
    84→
    85→  const style: React.CSSProperties = win.maximized
    86→    ? { position: 'absolute', top: 0, left: 0, right: 0, bottom: 40, zIndex: win.zIndex }
    87→    : { position: 'absolute', top: win.y, left: win.x, width: win.width, height: win.height, zIndex: win.zIndex }
    88→
    89→  return (
    90→    <div
    91→      ref={windowRef}
    92→      className={`xp-window ${isActive ? 'xp-window-active' : 'xp-window-inactive'}`}
    93→      style={style}
    94→      onMouseDown={() => focusWindow(win.id)}
    95→    >
    96→      {/* Title Bar */}
    97→      <div
    98→        className={`xp-titlebar ${isActive ? 'xp-titlebar-active' : 'xp-titlebar-inactive'}`}
    99→        onMouseDown={handleMouseDown}
   100→        onDoubleClick={() => win.maximized ? restoreWindow(win.id) : maximizeWindow(win.id)}
   101→      >
   102→        <div className="xp-titlebar-icon">
   103→          <XPAppIcon icon={win.icon} size={16} />
   104→        </div>
   105→        <div className={`xp-titlebar-text ${isActive ? '' : 'xp-titlebar-text-inactive'}`}>
   106→          {win.title}
   107→        </div>
   108→        <div className="xp-titlebar-buttons">
   109→          <button
   110→            className="xp-titlebar-btn xp-btn-minimize"
   111→            onClick={(e) => { e.stopPropagation(); minimizeWindow(win.id) }}
   112→            onMouseDown={(e) => e.stopPropagation()}
   113→          >
   114→            <MinimizeIcon />
   115→          </button>
   116→          <button
   117→            className="xp-titlebar-btn xp-btn-maximize"
   118→            onClick={(e) => { e.stopPropagation(); win.maximized ? restoreWindow(win.id) : maximizeWindow(win.id) }}
   119→            onMouseDown={(e) => e.stopPropagation()}
   120→          >
   121→            {win.maximized ? <RestoreDownIcon /> : <MaximizeIcon />}
   122→          </button>
   123→          <button
   124→            className="xp-titlebar-btn xp-btn-close"
   125→            onClick={(e) => { e.stopPropagation(); closeWindow(win.id) }}
   126→            onMouseDown={(e) => e.stopPropagation()}
   127→          >
   128→            <CloseIcon />
   129→          </button>
   130→        </div>
   131→      </div>
   132→
   133→      {/* Menu Bar */}
   134→      <div className="xp-menubar">
   135→        <span className="xp-menubar-item">File</span>
   136→        <span className="xp-menubar-item">Edit</span>
   137→        <span className="xp-menubar-item">View</span>
   138→        <span className="xp-menubar-item">Help</span>
   139→      </div>
   140→
   141→      {/* Content Area */}
   142→      <div className="xp-window-content">
   143→        <WindowContent content={win.content} />
   144→      </div>
   145→
   146→      {/* Status Bar */}
   147→      <div className="xp-statusbar">
   148→        <span className="xp-statusbar-text">Ready</span>
   149→      </div>
   150→
   151→      {/* Resize Handle */}
   152→      {!win.maximized && (
   153→        <div
   154→          className="xp-resize-handle"
   155→          onMouseDown={handleResizeStart}
   156→        />
   157→      )}
   158→    </div>
   159→  )
   160→}
